2번 예시 정답
class Calculator {
   int add(int a, int b) {
       return a + b;
   }

   int subtract(int a, int b) {
       return a - b;
   }

   int multiply(int a, int b) {
       return a * b;
   }

   double divide(int a, int b) {
       if (b == 0) {
           return 0;
       }
       return (double) a / b;
   }
}

public class Main2 {
   public static void main(String[] args) {
       Calculator c = new Calculator();
       System.out.println(c.add(10, 5));
       System.out.println(c.subtract(10, 5));
       System.out.println(c.multiply(10, 5));
       System.out.println(c.divide(10, 5));
   }
}

7번 예시 정답
abstract class Shape {
   abstract double area();
}

class Rectangle extends Shape {
   double width;
   double height;

   Rectangle(double width, double height) {
       this.width = width;
       this.height = height;
   }

   @Override
   double area() {
       return width * height;
   }
}

class Circle extends Shape {
   double radius;

   Circle(double radius) {
       this.radius = radius;
   }

   @Override
   double area() {
       return 3.14 * radius * radius;
   }
}

public class Main7 {
   public static void main(String[] args) {
       Shape[] shapes = new Shape[2];
       shapes[0] = new Rectangle(4, 5);
       shapes[1] = new Circle(3);

       for (Shape s : shapes) {
           System.out.println("넓이: " + s.area());
       }
   }
}

8번 예시 정답
interface Movable {
   void move(int x, int y);
}

class Car implements Movable {
   int x;
   int y;

   Car(int x, int y) {
       this.x = x;
       this.y = y;
   }

   @Override
   public void move(int x, int y) {
       this.x = x;
       this.y = y;
       System.out.println("자동차가 (" + x + ", " + y + ") 위치로 이동했습니다");
   }
}

public class Main8 {
   public static void main(String[] args) {
       Movable m = new Car(0, 0);
       m.move(10, 20);
   }
}

10번 예시 정답
class BankAccount {
   String owner;
   int balance;

   BankAccount(String owner, int balance) {
       this.owner = owner;
       this.balance = balance;
   }

   void deposit(int amount) {
       balance += amount;
       System.out.println(amount + "원 입금, 현재 잔액: " + balance);
   }

   void withdraw(int amount) {
       if (balance >= amount) {
           balance -= amount;
           System.out.println(amount + "원 출금, 현재 잔액: " + balance);
       } else {
           System.out.println("잔액 부족");
       }
   }
}

class SavingsAccount extends BankAccount {

   SavingsAccount(String owner, int balance) {
       super(owner, balance);
   }

   @Override
   void withdraw(int amount) {
       int max = balance / 2;
       if (amount > max) {
           System.out.println("적금 계좌는 한번에 잔액의 50%만 출금 가능합니다");
           return;
       }
       super.withdraw(amount);
   }
}

public class Main10 {
   public static void main(String[] args) {
       SavingsAccount acc = new SavingsAccount("홍길동", 100000);
       acc.withdraw(30000);
       System.out.println("현재 잔액: " + acc.balance);
       acc.withdraw(80000);
       System.out.println("현재 잔액: " + acc.balance);
   }
}
